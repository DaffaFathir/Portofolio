# Set timezone ke WIB
jakarta_tz = pytz.timezone('Asia/Jakarta')

def get_current_time():
    """Get current time in Jakarta timezone"""
    return datetime.now(jakarta_tz)

def parse_time(time_str):
    """Parse time string to datetime object in Jakarta timezone"""
    try:
        # Remove 'Z' if present and add UTC offset
        if time_str.endswith('Z'):
            time_str = time_str[:-1] + '+00:00'
        elif not ('+' in time_str or '-' in time_str):
            time_str = time_str + '+00:00'
            
        # Parse the time string
        dt = datetime.fromisoformat(time_str)
        
        # If no timezone info, assume UTC
        if dt.tzinfo is None:
            dt = pytz.UTC.localize(dt)
            
        # Convert to Jakarta timezone and subtract 7 hours
        jakarta_time = dt.astimezone(jakarta_tz)
        adjusted_time = jakarta_time - timedelta(hours=7)
        return adjusted_time
    except Exception as e:
        logger.error(f"Error parsing time: {e}")
        return get_current_time()

# Fungsi untuk memeriksa apakah pengguna adalah admin grup
async def is_user_admin(update: Update, context: CallbackContext) -> bool:
    try:
        # Dapatkan info chat member
        chat_member = await context.bot.get_chat_member(
            chat_id=GROUP_ID,
            user_id=update.effective_user.id
        )
        # Cek status admin
        return chat_member.status in ['creator', 'administrator']
    except Exception as e:
        logger.error(f"Error checking admin status: {e}")
        return False

# Fungsi wrapper untuk command yang memerlukan akses admin
def admin_only(func):
    async def wrapper(update: Update, context: CallbackContext):
        if not await is_user_admin(update, context):
            await update.message.reply_text(
                "⛔ Maaf, hanya admin grup yang dapat menggunakan perintah ini."
            )
            return
        return await func(update, context)
    return wrapper

# Fungsi untuk validasi pengiriman pesan ke grup
async def send_message_to_group(bot, message, parse_mode=None):

            chat_id=GROUP_ID,

    except Exception as e:
        logger.error(f"Error sending message to group: {e}")
        # Coba kirim tanpa parse_mode jika error parsing

                    text=message
                )

            except Exception as e2:
                logger.error(f"Error sending plain message to group: {e2}")
        return False

# Fungsi untuk mengirim notifikasi otomatis
async def send_automatic_alert(application, water_level, reading_time):
    global last_alert_level, last_notification_time, last_siaga_status

    logger.info(f"Checking alert for water level: {water_level}cm")

    # Tentukan status siaga berdasarkan ketinggian air
    current_siaga_status = None
    if water_level > 400:
        current_siaga_status = 1
    elif water_level >= 300:
        current_siaga_status = 2
    elif water_level >= 200:
        current_siaga_status = 3
    else:
        current_siaga_status = 4

    # Cek perubahan status siaga
    if last_siaga_status is not None and last_siaga_status != current_siaga_status:
        await send_status_change_notification(
            application,
            water_level,
            last_siaga_status,
            current_siaga_status
        )

    # Update status siaga terakhir
    last_siaga_status = current_siaga_status

    # Hanya kirim notifikasi detail jika water_level >= 300cm
    if water_level < 300:
        logger.info(f"Water level {water_level}cm is below 300cm threshold, no detailed alert needed")
        return

    alert_info = get_alert_info(water_level)
    current_time = get_current_time()

    logger.info(f"Alert level {alert_info['level']} triggered for {water_level}cm")

    # Cek cooldown - jangan spam notifikasi level yang sama dalam 1 menit
    if (last_alert_level == alert_info["level"] and 
        last_notification_time and 
        (current_time - last_notification_time).seconds < notification_cooldown):
        logger.info(f"Cooldown active for level {alert_info['level']}, skipping notification")
        return

    try:
        # Log raw time untuk debugging
        logger.info(f"Raw time from DB: {reading_time}")

        # Parse waktu dari string ISO format dan konversi ke WIB
        dt = parse_time(reading_time)
        formatted_time = dt.strftime("%H:%M WIB")

        logger.info(f"Formatted time: {formatted_time}")

        # Format pesan notifikasi untuk perubahan status
        alert_message = (
            f"🚨 *PERINGATAN DINI BANJIR - NOTIFIKASI PERUBAHAN*\n\n"
            f"⚠️ *Level Siaga:* {alert_info['status']}\n"
            f"⏰ *Waktu Pemantauan:* {formatted_time}\n"
            f"📍 *Lokasi:* Daerah Sekitar Sungai Sukapura\n"
            f"💧 *Ketinggian Air:* {water_level:.0f} cm\n\n"
            f"⚡ *TINDAKAN YANG HARUS DILAKUKAN:*\n{alert_info['tindakan']}\n\n"
            f"🔔 *PERINGATAN PENTING:*\n{alert_info['peringatan']}\n\n"
            f"❗ Harap selalu waspada dan ikuti instruksi dari petugas.\n\n"
            f"🌐 Silahkan kunjungi website untuk informasi yang lebih lanjut: https://earlywarningsystem.netlify.app/"
        )

        # Kirim ke grup menggunakan fungsi helper
        success = await send_message_to_group(application.bot, alert_message, parse_mode='Markdown')

        if success:
            # Update tracking variables
            last_alert_level = alert_info["level"]
            last_notification_time = current_time
            logger.info(f"✅ Alert sent successfully: Level {alert_info['level']} - {water_level}cm at {formatted_time}")
        else:
            # Coba kirim pesan sederhana jika pesan format gagal
            simple_message = (
                f"🚨 PERINGATAN BANJIR!\n"
                f"Tinggi air: {water_level}cm\n"
                f"Status: {alert_info['status']}\n"
                f"Waktu: {formatted_time}\n"
                f"Lokasi: Sungai Sukapura"
            )
            simple_success = await send_message_to_group(application.bot, simple_message)
            if simple_success:
                last_alert_level = alert_info["level"]
                last_notification_time = current_time
                logger.info("✅ Simple alert sent successfully")

    except Exception as e:
        logger.error(f"Error in alert: {e}")
        # Jika ada error dengan waktu, tetap coba kirim alert dengan waktu server
        try:
            fallback_time = get_current_time().strftime("%H:%M WIB")
            simple_message = f"🚨 PERINGATAN BANJIR!\nTinggi air: {water_level}cm\nStatus: {alert_info['level']}\nWaktu: {fallback_time}\nLokasi: Sungai Sukapura"
            await send_message_to_group(application.bot, simple_message)
        except Exception as e2:
            logger.error(f"Error sending fallback alert: {e2}")

def group_readings_by_30min(readings):
    groups = defaultdict(list)

    for reading in readings:
        try:
            water_level = float(reading["water_level"])

            # Konversi waktu ke WIB menggunakan helper function
            local_time = convert_to_wib(reading["reading_time"])

            # Bulatkan ke interval 30 menit terdekat
            minutes = local_time.minute
            if minutes < 30:
                rounded_time = local_time.replace(minute=0, second=0, microsecond=0)
                end_time = local_time.replace(minute=30, second=0, microsecond=0)
            else:
                rounded_time = local_time.replace(minute=30, second=0, microsecond=0)
                end_time = (local_time + timedelta(hours=1)).replace(minute=0, second=0, microsecond=0)

            time_key = (rounded_time, end_time)
            groups[time_key].append(water_level)
        except (ValueError, KeyError) as e:
            logger.error(f"Error processing reading: {e}")
            continue

    return groups

async def get_todays_readings():
    try:
        # Set timezone ke WIB
        jakarta_tz = pytz.timezone("Asia/Jakarta")
        now = datetime.now(jakarta_tz)
        today_start = now.replace(hour=0, minute=0, second=0, microsecond=0)

        # Konversi ke UTC untuk query
        utc_start = today_start.astimezone(pytz.UTC)
        today_start_utc = utc_start.isoformat()

        # Query data hari ini
        headers = {
            "apikey": SUPABASE_KEY,
            "Authorization": f"Bearer {SUPABASE_KEY}",
        }
        url = f"{SUPABASE_URL}/rest/v1/{TABLE_NAME}"
        params = {
            "select": "water_level,reading_time",
            "reading_time": f"gte.{today_start_utc}",
            "order": "reading_time.asc"
        }

        response = requests.get(url, headers=headers, params=params)
        response.raise_for_status()
        return response.json()
    except Exception as e:
        logger.error(f"Error getting today's readings: {e}")
        return []

async def report(update: Update, context: CallbackContext):
    try:
        # Ambil data hari ini
        readings = await get_todays_readings()

        if not readings:
            await update.message.reply_text("❌ Tidak ada data untuk hari ini.")
            return

        # Hitung statistik keseluruhan
        all_levels = [float(r["water_level"]) for r in readings]
        avg_level = statistics.mean(all_levels)
        max_level = max(all_levels)
        min_level = min(all_levels)

        # Grup data per 30 menit
        grouped_data = group_readings_by_30min(readings)

        # Format laporan
        report_text = "📊 *Laporan Tinggi Air Hari Ini:*\n"
        report_text += f"• Rata-rata: {avg_level:.2f} cm\n"
        report_text += f"• Tertinggi: {max_level:.1f} cm\n"
        report_text += f"• Terendah: {min_level:.1f} cm\n\n"

        report_text += "📌 *Data Per 30 Menit:*\n"

        # Sort groups by time in ascending order
        sorted_groups = sorted(grouped_data.items(), key=lambda x: x[0][0])

        # Kelompokkan data per hari
        current_day = None
        group_count = 0

        for (start_time, end_time), levels in sorted_groups:
            # Cek apakah ini hari yang berbeda
            if current_day != start_time.date():
                current_day = start_time.date()
                if group_count > 0:  # Tambah baris kosong antara hari yang berbeda
                    report_text += "\n"
                report_text += f"📅 *{current_day.strftime('%d %B %Y')}*\n"

            avg_group_level = statistics.mean(levels)
            alert_info = get_alert_info(avg_group_level)

            # Format waktu dalam WIB
            time_range = f"{start_time.strftime('%H:%M')}-{end_time.strftime('%H:%M')}"

            # Tentukan status icon
            if avg_group_level >= 400:
                status_icon = "🚨"  # Siaga 1
            elif avg_group_level >= 300:
                status_icon = "⚠️"  # Siaga 2
            elif avg_group_level >= 200:
                status_icon = "⚠️"  # Siaga 3
            elif avg_group_level >= 100:
                status_icon = "ℹ️"  # Siaga 4
            else:
                status_icon = "✅"  # Aman

            group_count += 1
            report_text += f"{group_count}. {time_range} - {avg_group_level:.1f} cm ({len(levels)} data) {status_icon} {alert_info['status']}\n"

            # Batasi jumlah baris yang ditampilkan
            if group_count >= 48:  # 24 jam x 2 (30 menit interval)
                break

        # Tambahkan link website
        report_text += "\n🌐 Silahkan kunjungi website untuk informasi yang lebih lanjut: https://earlywarningsystem.netlify.app/"
        
        # Kirim laporan
        await update.message.reply_markdown(report_text)

    except Exception as e:
        logger.error(f"Error generating report: {e}")
        await update.message.reply_text("❌ Terjadi kesalahan saat membuat laporan.")

            elif avg_group_level >= 300:
                status_icon = "⚠️"  # Siaga 2
            elif avg_group_level >= 200:
                status_icon = "⚠️"  # Siaga 3
            elif avg_group_level >= 100:
                status_icon = "ℹ️"  # Siaga 4
            else:
                status_icon = "✅"  # Aman

            group_count += 1
            history_text += f"{group_count}. {time_range} - {avg_group_level:.1f} cm ({len(levels)} data) {status_icon} {alert_info['status']}\n"

            # Batasi jumlah baris yang ditampilkan
            if group_count >= 48:  # 24 jam x 2 (30 menit interval)
                break

        # Tambahkan link website
        history_text += "\n🌐 Silahkan kunjungi website untuk informasi yang lebih lanjut: https://earlywarningsystem.netlify.app/"
        
        # Kirim laporan
        await update.message.reply_markdown(history_text)

    except Exception as e:
        logger.error(f"Error generating history: {e}")
        await update.message.reply_text("❌ Terjadi kesalahan saat mengambil riwayat data.")

# Fungsi untuk mengirim update status per jam ke grup
async def send_hourly_status_update(application):
    global last_hourly_update

    current_time = get_current_time()

    # Cek apakah sudah waktunya untuk update (setiap jam)
    if (last_hourly_update is not None and 
        (current_time - last_hourly_update).seconds < 3600):  # 3600 detik = 1 jam
        return

    try:
        # Ambil data terkini dari database
        headers = {
            "apikey": SUPABASE_KEY,
            "Authorization": f"Bearer {SUPABASE_KEY}",
        }
        url = f"{SUPABASE_URL}/rest/v1/{TABLE_NAME}?order=reading_time.desc&limit=1"

        response = requests.get(url, headers=headers)
        response.raise_for_status()
        data = response.json()

        if data and len(data) > 0:
            latest = data[0]
            water_level = float(latest["water_level"])
            alert_info = get_alert_info(water_level)

            # Format waktu WIB
            reading_time = latest["reading_time"]
            dt = parse_time(reading_time)
            formatted_time = dt.strftime("%H:%M WIB")

            # Tentukan emoji status
            status_emoji = "✅"  # Default untuk kondisi normal
            if water_level > 400:
                status_emoji = "🚨"  # Siaga 1
            elif water_level >= 300:
                status_emoji = "⚠️"  # Siaga 2
            elif water_level >= 200:
                status_emoji = "⚠️"  # Siaga 3

            # Format pesan status untuk update rutin per jam
            status_message = (
                f"📊 *LAPORAN RUTIN KETINGGIAN AIR*\n\n"
                f"═══════════════════\n"
                f"🕒 *Waktu Update:* {formatted_time}\n"
                f"📍 *Lokasi:* Sungai Sukapura\n"
                f"═══════════════════\n\n"
                f"📈 *INFORMASI KETINGGIAN:*\n"
                f"• Tinggi Air: {water_level:.1f} cm\n"
                f"• Status: {status_emoji} {alert_info['status']}\n\n"
                f"💡 *Catatan:*\n{alert_info['peringatan']}\n\n"
                f"═══════════════════\n"
                f"ℹ️ Update rutin berikutnya dalam 1 jam\n"
                f"Tetap waspada dan pantau perubahan ketinggian air\n\n"
                f"🌐 Silahkan kunjungi website untuk informasi yang lebih lanjut: https://earlywarningsystem.netlify.app/"
            )

            # Kirim pesan ke grup
            await send_message_to_group(application.bot, status_message, parse_mode='Markdown')
            logger.info(f"Hourly status update sent at {formatted_time}")

            # Update waktu terakhir update
            last_hourly_update = current_time

        else:
            logger.warning("No data available for hourly update")

    except Exception as e:
        logger.error(f"Error sending hourly status update: {e}")

# Fungsi loop monitoring yang berjalan setiap 30 detik
async def monitoring_loop(application):
    while True:  # Ubah ke while True karena monitoring selalu aktif
        try:
            # Monitoring ketinggian air
            await monitor_water_level(application)

            # Kirim update status per jam
            await send_hourly_status_update(application)

            await asyncio.sleep(30)
        except asyncio.CancelledError:
            logger.info("Monitoring task cancelled")
            break
        except Exception as e:
            logger.error(f"Error in monitoring loop: {e}")
            await asyncio.sleep(30)  # Tetap tunggu sebelum mencoba lagi

async def main():
    logger.info("🤖 Starting Telegram Bot...")
    application = Application.builder().token(BOT_TOKEN).build()

    # Tambahkan command handlers
    application.add_handler(CommandHandler("help", help_command))
    application.add_handler(CommandHandler("status", status))
    application.add_handler(CommandHandler("report", report))
    application.add_handler(CommandHandler("history", history))

    # Handler untuk menyambut anggota baru
    application.add_handler(MessageHandler(filters.StatusUpdate.NEW_CHAT_MEMBERS, welcome_new_member))

    # Handler untuk pesan yang tidak dikenal
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, unknown_message))

    try:
        logger.info("📱 Starting bot polling...")
        await application.initialize()
        await application.updater.start_polling()
        await application.start()

        # Mulai monitoring secara otomatis
        global monitoring_task, last_hourly_update
        last_hourly_update = get_current_time()  # Inisialisasi waktu update terakhir
        monitoring_task = asyncio.create_task(monitoring_loop(application))
        logger.info("🚀 Monitoring task started automatically")

        # Kirim notifikasi ke grup bahwa bot telah aktif
        await application.bot.send_message(
            chat_id=GROUP_ID,
            text="🔔 Bot Early Warning System telah aktif dan mulai monitoring secara otomatis!\n"
                 "ℹ️ Status ketinggian air akan dikirim setiap 1 jam sekali.\n\n"
                 "🌐 Silahkan kunjungi website untuk informasi yang lebih lanjut: https://earlywarningsystem.netlify.app/"
        )

        # Tunggu sampai program dihentikan
        while True:
            await asyncio.sleep(1)

    except Exception as e:
        logger.error(f"Error in main loop: {e}")
    finally:
        # Pastikan semua task dibersihkan dengan benar
        logger.info("Shutting down...")

        # Hentikan monitoring jika masih berjalan
        if monitoring_task and not monitoring_task.done():
            monitoring_task.cancel()
            try:
                await monitoring_task
            except asyncio.CancelledError:
                pass

        try:
            if application.updater.running:
                await application.updater.stop()
            if application.running:
                await application.stop()
            await application.shutdown()
        except Exception as e:
            logger.error(f"Error stopping application: {e}")

if __name__ == '__main__':
    try:
        asyncio.run(main())
    except KeyboardInterrupt:
        logger.info("Bot stopped by user")
    except Exception as e:
        logger.error(f"Error running bot: {e}")
