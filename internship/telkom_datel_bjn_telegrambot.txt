# Fungsi untuk mengautentikasi dan menginisialisasi client gspread
def init_gspread():
    if not os.path.isfile(CREDENTIALS_PATH):
        logging.error(f"Credentials file not found: {CREDENTIALS_PATH}")
        return None

    try:
        scope = ["https://spreadsheets.google.com/feeds", "https://www.googleapis.com/auth/drive"]
        creds = ServiceAccountCredentials.from_json_keyfile_name(CREDENTIALS_PATH, scope)
        client = gspread.authorize(creds)
        return client
    except Exception as e:
        logging.error(f"Error initializing gspread client: {e}")
        return None

# Fungsi untuk mengambil data dari Google Sheets
def get_sheet_data(sheet_name, range_name="A1:AX1000000"):
    client = init_gspread()
    if not client:
        logging.error("Google Sheets client initialization failed")
        return None
    try:
        sheet = client.open_by_key(SHEET_ID).worksheet(sheet_name)
        data = sheet.get(range_name)
        return data
    except gspread.exceptions.WorksheetNotFound:
        logging.error(f"Worksheet '{sheet_name}' not found")
        return None
    except Exception as e:
        logging.error(f"Error getting data from Google Sheets: {e}")
        return None

def load_data_from_cache():
    if os.path.exists(CACHE_FILE):
        with open(CACHE_FILE, 'r') as file:
            try:
                data = json.load(file)
                logging.info("Data loaded from cache.")
                return data['content']
            except (ValueError, KeyError):
                logging.error("Error reading cache.")
    return None

def save_data_to_cache(data):
    with open(CACHE_FILE, 'w') as file:
        json.dump({'cache_time': datetime.now().isoformat(), 'content': data}, file)
    logging.info("Data saved to cache.")
    print("Cache telah diperbarui.")

def send_telegram_message(chat_id, text, forward_to_group=False):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    payload = {
        "chat_id": chat_id,
        "text": text,
        "parse_mode": "HTML"
    }
    try:
        response = requests.post(url, json=payload, timeout=60)
        logging.info(f"Sending message to {chat_id}: {text}")
        logging.info(f"Response status code: {response.status_code}")
        
        if response.status_code == 429:
            retry_after = int(response.headers.get("Retry-After", 5))  # Ambil waktu retry dari header jika ada
            logging.warning(f"Rate limit hit. Retrying after {retry_after} seconds.")
            time.sleep(retry_after)  # Tunggu sesuai waktu retry yang disarankan
            send_telegram_message(chat_id, text, forward_to_group)  # Coba kirim ulang pesan
        elif response.status_code != 200:
            logging.error(f"Failed to send message: {response.text}")
            return

        try:
            logging.info(f"Response JSON: {response.json()}")
        except ValueError:
            logging.error(f"Response text is not valid JSON: {response.text}")

        if forward_to_group:
            forward_message_to_group(text)

    except Exception as e:
        logging.error(f"Error sending message: {e}")

def forward_message_to_group(text):
    send_telegram_message(FORWARD_GROUP_ID, text)
    logging.info(f"Forwarded message to group {FORWARD_GROUP_ID}: {text}")

# Fungsi untuk memperbarui nilai sel di spreadsheet
def update_status_to_close(chat_id, identifier, close_type):
    client = init_gspread()
    if not client:
        bot.send_message(chat_id, "Error initializing Google Sheets client.")
        return

    try:
        sheet = client.open_by_key(SHEET_ID).worksheet("23")
        today = datetime.now().date()  # Get today's date as a date object

        if close_type.upper() == "UNSPEC":
            cells = sheet.findall(identifier, in_column=9)  # Search for serviceno in column 9
            updated = False
            for cell in cells:
                row = sheet.row_values(cell.row)  # Get all values in the row
                close_date_str = f"{row[24]}-{row[45]}-{row[44]}"  # Format date from columns 25, 46, and 45
                logging.info(f"Raw close date string: {close_date_str}")

                try:
                    close_date = datetime.strptime(close_date_str, "%d-%m-%Y").date()
                    logging.info(f"Parsed close date: {close_date}")
                except ValueError as e:
                    logging.error(f"Error parsing date string '{close_date_str}': {e}")
                    continue

                # Check if the status is 'UNSPEC' and the date matches today
                current_status = row[13]  # Assuming status is in the 14th column (index 13)
                if close_date == today and current_status == "⚡️UNSPEC":
                    col = 14  # Status column at index 13 (14th column)
                    close_status = f"CLOSE UNSPEC"
                    sheet.update_cell(cell.row, col, close_status)
                    bot.send_message(chat_id, f"Status untuk service number '{identifier}' telah diperbarui menjadi '{close_status}'.")
                    updated = True
            if not updated:
                bot.send_message(chat_id, f"Service number '{identifier}' tidak ditemukan atau status bukan 'UNSPEC'.")
        elif close_type.upper() in ["TTR", "SQM"]:
            cells = sheet.findall(identifier, in_column=1)  # Assuming Incident is in column 1
            updated = False
            for cell in cells:
                row = sheet.row_values(cell.row)  # Get all values in the row
                current_status = row[13]  # Assuming status is in the 14th column (index 13)

                if close_type.upper() == "TTR" and current_status == "🔥TTR":
                    col = 14  # Status column at index 13 (14th column)
                    close_status = "CLOSE TTR"
                    sheet.update_cell(cell.row, col, close_status)
                    bot.send_message(chat_id, f"Status untuk Incident '{identifier}' telah diperbarui menjadi '{close_status}'.")
                    updated = True

                elif close_type.upper() == "SQM" and current_status == "💊SQM":
                    col = 14  # Status column at index 13 (14th column)
                    close_status = "CLOSE SQM"
                    sheet.update_cell(cell.row, col, close_status)
                    bot.send_message(chat_id, f"Status untuk Incident '{identifier}' telah diperbarui menjadi '{close_status}'.")
                    updated = True

            if not updated:
                bot.send_message(chat_id, f"Incident '{identifier}' tidak ditemukan atau status tidak sesuai.")
    except Exception as e:
        bot.send_message(chat_id, f"Failed to update status for '{identifier}': {e}")
        logging.error(f"Exception occurred: {e}")

def send_produktif_summary(chat_id):
    data = get_sheet_data("23", "A1:AX1000000")
    if data is None:
        send_telegram_message(chat_id, "Error retrieving data from the spreadsheet.")
        return

    workzone_summary = {}
    current_time = datetime.now().strftime("%Y-%m-%d %H:%M")
    today = datetime.now().date()

    total_open_ttr = 0
    total_close_ttr = 0
    total_open_unspec = 0
    total_close_unspec = 0
    total_open_sqm = 0
    total_close_sqm = 0

    for row in data:
        try:
            if len(row) > 13:
                incident = row[0]
                workzone = row[16]
                nama = row[22]

                if workzone not in workzone_summary:
                    workzone_summary[workzone] = {'TTR': {}, 'UNSPEC': {}, 'SQM': {}}

                if "🔥TTR" in row[13]:
                    workzone_summary[workzone]['TTR'].setdefault(nama, {'open': 0, 'close': 0, 'total': 0})
                    workzone_summary[workzone]['TTR'][nama]['open'] += 1
                    total_open_ttr += 1

                elif "⚡️UNSPEC" in row[13]:
                    workzone_summary[workzone]['UNSPEC'].setdefault(nama, {'open': 0, 'close': 0, 'total': 0})
                    workzone_summary[workzone]['UNSPEC'][nama]['open'] += 1
                    total_open_unspec += 1

                elif "💊SQM" in row[13]:
                    workzone_summary[workzone]['SQM'].setdefault(nama, {'open': 0, 'close': 0, 'total': 0})
                    workzone_summary[workzone]['SQM'][nama]['open'] += 1
                    total_open_sqm += 1

            if len(row) > 30 and ("CLOSE TTR" in row[13] or "CLOSE UNSPEC" in row[13] or "CLOSE SQM" in row[13]):
                try:
                    close_date_str = f"{row[24]}-{row[45]}-{row[44]}"  # Sesuaikan urutan kolom dengan yang benar
                    close_date = datetime.strptime(close_date_str, "%d-%m-%Y").date()
                except ValueError:
                    logging.error(f"Error parsing date: {close_date_str}")
                    continue

                if close_date == today:
                    workzone = row[16]
                    nama = row[22]

                    if workzone not in workzone_summary:
                        workzone_summary[workzone] = {'TTR': {}, 'UNSPEC': {}, 'SQM': {}}

                    if nama not in workzone_summary[workzone]['TTR']:
                        workzone_summary[workzone]['TTR'][nama] = {'open': 0, 'close': 0, 'total': 0}
                    if nama not in workzone_summary[workzone]['UNSPEC']:
                        workzone_summary[workzone]['UNSPEC'][nama] = {'open': 0, 'close': 0, 'total': 0}
                    if nama not in workzone_summary[workzone]['SQM']:
                        workzone_summary[workzone]['SQM'][nama] = {'open': 0, 'close': 0, 'total': 0}

                    if "CLOSE TTR" in row[13]:
                        workzone_summary[workzone]['TTR'][nama]['close'] += 1
                        total_close_ttr += 1
                    elif "CLOSE UNSPEC" in row[13]:
                        workzone_summary[workzone]['UNSPEC'][nama]['close'] += 1
                        total_close_unspec += 1
                    elif "CLOSE SQM" in row[13]:
                        workzone_summary[workzone]['SQM'][nama]['close'] += 1
                        total_close_sqm += 1

        except IndexError as e:
            logging.error(f"Error processing row: {row}, error: {e}")

    summary_message = generate_summary_message(workzone_summary, current_time, total_open_ttr, total_close_ttr, total_open_unspec, total_close_unspec, total_open_sqm, total_close_sqm)
    send_telegram_message(chat_id, summary_message)

def run_repeater():
    while not stop_event.is_set():

def run_bot():
    while not stop_event.is_set():
        try:
            bot.polling(none_stop=True, interval=3)
        except Exception as e:
            logging.error(f"Error in run_bot: {e}")
            time.sleep(5)  # Tunggu sebelum mencoba lagi

def monitor_resources():
    while not stop_event.is_set():
        cpu_usage = psutil.cpu_percent(interval=1)
        memory_usage = psutil.virtual_memory().percent

        if cpu_usage > CPU_THRESHOLD:
            logging.warning(f"High CPU usage detected: {cpu_usage}%")
            # Kirim notifikasi atau tindakan lain jika diperlukan

        if memory_usage > MEMORY_THRESHOLD:
            logging.warning(f"High memory usage detected: {memory_usage}%")
            # Kirim notifikasi atau tindakan lain jika diperlukan

        time.sleep(5)  # Interval pengecekan

if __name__ == "__main__":
    if not load_data_from_cache():
        logging.info("Cache is empty or does not exist. Fetching new data.")
        update_cache()

    bot_thread = threading.Thread(target=run_bot)
    bot_thread.start()

    repeater_thread = threading.Thread(target=run_repeater)
    repeater_thread.start()

    monitor_thread = threading.Thread(target=monitor_resources)
    monitor_thread.start()

    cache_monitor_thread = threading.Thread(target=monitor_spreadsheet_updates)
    cache_monitor_thread.start()

    try:
        while bot_thread.is_alive() and repeater_thread.is_alive() and monitor_thread.is_alive() and cache_monitor_thread.is_alive():

            repeater_thread.join(1)
            monitor_thread.join(1)
            cache_monitor_thread.join(1)
    except KeyboardInterrupt:

        stop_event.set()
        bot_thread.join()
        repeater_thread.join()
        monitor_thread.join()
        cache_monitor_thread.join()
